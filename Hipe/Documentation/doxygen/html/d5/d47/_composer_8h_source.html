195  static Data getDataFromComposer(const std::string datatype, const boost::property_tree::ptree& dataNode)
196  {

198  switch (ioDataType)

203  case IODataType::VIDF:

205  return loadVideoFromFile(dataNode);
206  case IODataType::SEQIMGD:
207  filter::data::Composer::checkJsonFieldExist(dataNode, "path");
208  return loadImagesFromDirectory(dataNode.get<std::string>("path"));
209  case IODataType::STRMVID:
210  filter::data::Composer::checkJsonFieldExist(dataNode, "path");
211  return loadVideoFromStream(dataNode.get<std::string>("path"));
212  case IODataType::LISTIO:
213  filter::data::Composer::checkJsonFieldExist(dataNode, "array");
214  return loadListIoData(dataNode);

218  return loadPatternData(dataNode);
219  case IODataType::SQR_CROP:
