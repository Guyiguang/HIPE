246  bool parse_request(const std::shared_ptr<Request<socket_type>>& request) const
247  {
248  std::string line;
249  getline(request->content, line);
250  size_t method_end;
251  if ((method_end = line.find(' ')) != std::string::npos)
252  {
253  size_t path_end;
254  if ((path_end = line.find(' ', method_end + 1)) != std::string::npos)
255  {
256  request->method = line.substr(0, method_end);
257  request->path = line.substr(method_end + 1, path_end - method_end - 1);
258 
259  size_t protocol_end;
260  if ((protocol_end = line.find('/', path_end + 1)) != std::string::npos)
261  {
262  if (line.compare(path_end + 1, protocol_end - path_end - 1, "HTTP") != 0)
263  return false;
264  request->http_version = line.substr(protocol_end + 1, line.size() - protocol_end - 2);
265  }
266  else
267  return false;
268 
269  getline(request->content, line);
270  size_t param_end;
271  while ((param_end = line.find(':')) != std::string::npos)
272  {
273  size_t value_start = param_end + 1;
274  if ((value_start) < line.size())
275  {
276  if (line[value_start] == ' ')
277  value_start++;
278  if (value_start < line.size())
279  request->header.insert(std::make_pair(line.substr(0, param_end), line.substr(value_start, line.size() - value_start - 1)));
280  }
281 
282  getline(request->content, line);
283  }
284  }
285  else
286  return false;
287  }
288  else
289  return false;
290  return true;
